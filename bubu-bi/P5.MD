# P5 - 智能模式切换方案实施计划

## 项目概述

本文档详细规划了将现有的"AI对话"和"数据查询"两个独立TAB整合为智能模式切换方案的完整实施计划。该方案旨在提供无缝的用户体验，通过智能识别用户输入类型，在统一界面中实现对话式和查询式两种交互模式的自动切换。

## 核心设计理念

### 1. 智能识别与无缝切换
- **自动模式检测**：基于用户输入内容智能判断是自然语言对话还是SQL查询
- **统一交互界面**：保持对话式界面的连续性，避免模式切换的突兀感
- **上下文保持**：在模式切换过程中保持对话历史和查询上下文

### 2. 渐进式用户体验
- **零学习成本**：用户无需了解模式切换机制，系统自动处理
- **智能提示**：在适当时机提供模式切换的视觉反馈
- **灵活回退**：支持用户手动调整系统的智能判断

### 3. 🧠 AI思维链透明度
- **可视化推理过程**：实时展示AI的决策思维链，让用户了解系统如何分析和处理输入
- **置信度展示**：显示每个决策步骤的置信度，增强用户对系统判断的信任
- **交互式探索**：用户可以深入查看AI的推理细节，理解分析逻辑
- **学习价值**：通过观察AI思维过程，用户可以学习数据分析的最佳实践

### 1. 智能输入识别引擎

#### 1.1 SQL检测算法
```typescript
// utils/SmartInputDetector.ts
export class SmartInputDetector {
  private sqlKeywords = [
    'SELECT', 'FROM', 'WHERE', 'JOIN', 'GROUP BY', 'ORDER BY',
    'INSERT', 'UPDATE', 'DELETE', 'CREATE', 'ALTER', 'DROP'
  ];
  
  private sqlPatterns = [
    /^\s*SELECT\s+/i,
    /^\s*WITH\s+/i,
    /^\s*(INSERT|UPDATE|DELETE)\s+/i,
    /\bFROM\s+\w+/i,
    /\bWHERE\s+/i
  ];
  
  detectInputType(input: string): {
    type: 'sql' | 'natural' | 'mixed';
    confidence: number;
    suggestions?: string[];
  } {
    const sqlScore = this.calculateSQLScore(input);
    const naturalScore = this.calculateNaturalScore(input);
    
    if (sqlScore > 0.8) {
      return { type: 'sql', confidence: sqlScore };
    } else if (naturalScore > 0.7) {
      return { type: 'natural', confidence: naturalScore };
    } else {
      return { 
        type: 'mixed', 
        confidence: Math.max(sqlScore, naturalScore),
        suggestions: this.generateSuggestions(input)
      };
    }
  }
  
  private calculateSQLScore(input: string): number {
    let score = 0;
    const normalizedInput = input.toUpperCase();
    
    // 关键词匹配
    this.sqlKeywords.forEach(keyword => {
      if (normalizedInput.includes(keyword)) {
        score += 0.2;
      }
    });
    
    // 模式匹配
    this.sqlPatterns.forEach(pattern => {
      if (pattern.test(input)) {
        score += 0.3;
      }
    });
    
    // 语法结构检测
    if (this.hasSQLStructure(input)) {
      score += 0.3;
    }
    
    return Math.min(score, 1.0);
  }
  
  private calculateNaturalScore(input: string): number {
    let score = 0;
    
    // 问句检测
    if (/^(什么|如何|为什么|哪些|多少|怎么|请|帮我)/i.test(input)) {
      score += 0.4;
    }
    
    // 自然语言模式
    if (/[，。？！；：]/g.test(input)) {
      score += 0.2;
    }
    
    // 业务术语检测
    const businessTerms = ['分析', '统计', '查看', '显示', '计算', '比较'];
    businessTerms.forEach(term => {
      if (input.includes(term)) {
        score += 0.1;
      }
    });
    
    return Math.min(score, 1.0);
  }
}
```

#### 1.2 统一查询处理器
```typescript
// hooks/useUnifiedQueryProcessor.ts
export const useUnifiedQueryProcessor = () => {
  const detector = new SmartInputDetector();
  const [processingMode, setProcessingMode] = useState<'auto' | 'manual'>('auto');
  
  const processQuery = async (input: string, forceMode?: 'sql' | 'natural') => {
    const detection = forceMode ? 
      { type: forceMode, confidence: 1.0 } : 
      detector.detectInputType(input);
    
    // 更新UI状态指示器
    setCurrentMode(detection.type);
    setModeConfidence(detection.confidence);
    
    try {
      switch (detection.type) {
        case 'sql':
          return await processSQLQuery(input);
        case 'natural':
          return await processNaturalLanguageQuery(input);
        case 'mixed':
          return await processMixedQuery(input, detection.suggestions);
      }
    } catch (error) {
      return handleQueryError(error, detection.type);
    }
  };
  
  const processSQLQuery = async (sql: string) => {
    // 直接执行SQL查询
    const result = await executeQuery(sql);
    
    // 生成对话式响应
    const conversationResponse = {
      type: 'sql_result',
      message: `查询执行完成，共找到 ${result.total} 条记录`,
      result,
      suggestions: generateFollowUpQuestions(result)
    };
    
    return conversationResponse;
  };
  
  const processNaturalLanguageQuery = async (query: string) => {
    // 自然语言处理流程
    const llmResult = await processNaturalLanguage(query);
    
    if (llmResult.sql) {
      const result = await executeQuery(llmResult.sql);
      return {
        type: 'natural_result',
        message: llmResult.explanation,
        result,
        sql: llmResult.sql,
        confidence: llmResult.confidence
      };
    }
    
    return {
      type: 'natural_response',
      message: llmResult.response
    };
  };
};
```

#### 2.1 SmartInputArea组件
```typescript
// components/SmartInputArea.tsx
interface SmartInputAreaProps {
  value: string;
  onChange: (value: string) => void;
  onSubmit: (query: string, type: 'sql' | 'natural' | 'mixed') => void;
  loading?: boolean;
  placeholder?: string;
  // 🧠 思维链相关属性
  thinkingSteps?: ThinkingStep[];
  currentThinkingStep?: number;
  showThinking?: boolean;
  onThinkingToggle?: (show: boolean) => void;
}

export const SmartInputArea: React.FC<SmartInputAreaProps> = ({
  value,
  onChange,
  onSubmit,
  loading,
  placeholder = "输入您的问题或SQL查询...",
  thinkingSteps,
  currentThinkingStep,
  showThinking = false,
  onThinkingToggle
}) => {
  const [inputType, setInputType] = useState<'sql' | 'natural' | 'mixed'>('natural');
  const [confidence, setConfidence] = useState(0);
  const [showThinkingLocal, setShowThinkingLocal] = useState(showThinking);
  const [showModeSelector, setShowModeSelector] = useState(false);
  const detector = useMemo(() => new SmartInputDetector(), []);
  
  // 思维链显示状态管理
  const handleThinkingToggle = () => {
    const newShow = !showThinkingLocal;
    setShowThinkingLocal(newShow);
    onThinkingToggle?.(newShow);
  };
  
  // 同步外部showThinking状态
  useEffect(() => {
    setShowThinkingLocal(showThinking);
  }, [showThinking]);
  
  const handleInputChange = (newValue: string) => {
    onChange(newValue);
    
    // 实时检测输入类型
    const detection = detector.detectInputType(newValue);
    setInputType(detection.type);
    setConfidence(detection.confidence);
  };
  
  const getModeIndicator = () => {
    const indicators = {
      sql: { icon: '⌨️', label: 'SQL查询', color: 'text-blue-500' },
      natural: { icon: '💬', label: '自然语言', color: 'text-green-500' },
      mixed: { icon: '🤔', label: '混合模式', color: 'text-yellow-500' }
    };
    
    const indicator = indicators[inputType];
    
    return (
      <div className={`flex items-center space-x-1 ${indicator.color}`}>
        <span>{indicator.icon}</span>
        <span className="text-xs">{indicator.label}</span>
        <div className="w-12 h-1 bg-gray-200 rounded">
          <div 
            className="h-1 bg-current rounded transition-all duration-300"
            style={{ width: `${confidence * 100}%` }}
          />
        </div>
      </div>
    );
  };
  
  return (
    <div className="smart-input-container">
      {/* 模式指示器 */}
      <div className="flex justify-between items-center mb-2">
        {getModeIndicator()}
        <button 
          className="text-xs text-gray-500 hover:text-gray-700"
          onClick={() => setShowModeSelector(!showModeSelector)}
        >
          手动选择模式
        </button>
      </div>
      
      {/* 输入区域 */}
      <div className="relative">
        <textarea
          value={value}
          onChange={(e) => handleInputChange(e.target.value)}
          placeholder={placeholder}
          className="w-full p-3 border rounded-lg resize-none focus:ring-2 focus:ring-blue-500"
          rows={3}
        />
        
        {/* 智能提示 */}
        {inputType === 'mixed' && (
          <div className="absolute top-2 right-2">
            <div className="bg-yellow-100 text-yellow-800 text-xs px-2 py-1 rounded">
              需要澄清查询类型
            </div>
          </div>
        )}
      </div>
      
      {/* 提交按钮 */}
      <div className="flex justify-between items-center mt-3">
        <div className="text-xs text-gray-500">
          {inputType === 'sql' && '将直接执行SQL查询'}
          {inputType === 'natural' && '将转换为SQL后执行'}
          {inputType === 'mixed' && '系统将智能判断处理方式'}
        </div>
        <div className="flex items-center space-x-2">
          {/* 思维链切换按钮 */}
          <button
            onClick={handleThinkingToggle}
            className={`px-3 py-1 text-xs rounded transition-colors ${
              showThinkingLocal 
                ? 'bg-purple-100 text-purple-700 hover:bg-purple-200' 
                : 'bg-gray-100 text-gray-600 hover:bg-gray-200'
            }`}
            title="显示/隐藏AI思维过程"
          >
            🧠 {showThinkingLocal ? '隐藏思维' : '显示思维'}
          </button>
          <button
            onClick={() => onSubmit(value, inputType)}
            disabled={loading || !value.trim()}
            className="px-4 py-2 bg-blue-500 text-white rounded hover:bg-blue-600 disabled:opacity-50"
          >
            {loading ? '处理中...' : '发送'}
          </button>
        </div>
      </div>
      
      {/* AI思维链展示 */}
      {showThinkingLocal && thinkingSteps && thinkingSteps.length > 0 && (
        <div className="mt-4">
          <EnhancedThinkingVisualization
            steps={thinkingSteps}
            currentStep={currentThinkingStep}
            mode="compact"
            showDetails={false}
          />
        </div>
      )}
    </div>
  );
};
```

#### 3.1 EnhancedThinkingVisualization组件
```typescript
// components/EnhancedThinkingVisualization.tsx
interface ThinkingStep extends ProcessingStep {
  reasoning?: string;           // 推理过程描述
  confidence?: number;          // 置信度 (0-1)
  alternatives?: string[];      // 备选方案
  tokenUsage?: {
    input: number;
    output: number;
    total: number;
  };
  metadata?: {
    processingTime: number;
    modelUsed: string;
    temperature: number;
  };
}

interface ThinkingVisualizationProps {
  steps: ThinkingStep[];
  currentStep?: number;
  showDetails?: boolean;
  mode?: 'compact' | 'detailed' | 'interactive';
  onStepClick?: (step: ThinkingStep, index: number) => void;
}

export const EnhancedThinkingVisualization: React.FC<ThinkingVisualizationProps> = ({
  steps,
  currentStep = -1,
  showDetails = true,
  mode = 'detailed',
  onStepClick
}) => {
  const [expandedSteps, setExpandedSteps] = useState<Set<number>>(new Set());
  const [selectedStep, setSelectedStep] = useState<number | null>(null);

  const toggleStepExpansion = (index: number) => {
    const newExpanded = new Set(expandedSteps);
    if (newExpanded.has(index)) {
      newExpanded.delete(index);
    } else {
      newExpanded.add(index);
    }
    setExpandedSteps(newExpanded);
  };

  const renderConfidenceBar = (confidence: number) => (
    <div className="flex items-center space-x-2 mt-1">
      <span className="text-xs text-gray-500">置信度:</span>
      <div className="flex-1 bg-gray-200 rounded-full h-2 max-w-20">
        <div 
          className={`h-2 rounded-full transition-all duration-300 ${
            confidence > 0.8 ? 'bg-green-500' : 
            confidence > 0.6 ? 'bg-yellow-500' : 'bg-red-500'
          }`}
          style={{ width: `${confidence * 100}%` }}
        />
      </div>
      <span className="text-xs font-medium">{Math.round(confidence * 100)}%</span>
    </div>
  );

  const renderThinkingDetails = (step: ThinkingStep, index: number) => {
    if (!expandedSteps.has(index)) return null;
    
    return (
      <div className="mt-3 p-3 bg-gray-50 rounded-lg border-l-4 border-blue-200">
        {/* 推理过程 */}
        {step.reasoning && (
          <div className="mb-3">
            <h5 className="text-sm font-medium text-gray-700 mb-1">🤔 推理过程</h5>
            <p className="text-sm text-gray-600 leading-relaxed">{step.reasoning}</p>
          </div>
        )}
        
        {/* 备选方案 */}
        {step.alternatives && step.alternatives.length > 0 && (
          <div className="mb-3">
            <h5 className="text-sm font-medium text-gray-700 mb-1">🔄 备选方案</h5>
            <ul className="text-sm text-gray-600 space-y-1">
              {step.alternatives.map((alt, i) => (
                <li key={i} className="flex items-start space-x-2">
                  <span className="text-gray-400 mt-0.5">•</span>
                  <span>{alt}</span>
                </li>
              ))}
            </ul>
          </div>
        )}
        
        {/* Token使用统计 */}
        {step.tokenUsage && (
          <div className="mb-3">
            <h5 className="text-sm font-medium text-gray-700 mb-1">📊 Token使用</h5>
            <div className="grid grid-cols-3 gap-2 text-xs">
              <div className="bg-blue-100 p-2 rounded text-center">
                <div className="font-medium text-blue-800">{step.tokenUsage.input}</div>
                <div className="text-blue-600">输入</div>
              </div>
              <div className="bg-green-100 p-2 rounded text-center">
                <div className="font-medium text-green-800">{step.tokenUsage.output}</div>
                <div className="text-green-600">输出</div>
              </div>
              <div className="bg-purple-100 p-2 rounded text-center">
                <div className="font-medium text-purple-800">{step.tokenUsage.total}</div>
                <div className="text-purple-600">总计</div>
              </div>
            </div>
          </div>
        )}
        
        {/* 元数据 */}
        {step.metadata && (
          <div className="text-xs text-gray-500 space-y-1">
            <div>⏱️ 处理时间: {step.metadata.processingTime}ms</div>
            <div>🤖 模型: {step.metadata.modelUsed}</div>
            <div>🌡️ 温度: {step.metadata.temperature}</div>
          </div>
        )}
      </div>
    );
  };

  return (
    <div className="thinking-visualization bg-white rounded-lg border p-4">
      <div className="flex items-center justify-between mb-4">
        <h3 className="text-lg font-semibold text-gray-800 flex items-center space-x-2">
          <span>🧠</span>
          <span>AI思维过程</span>
        </h3>
        <div className="text-sm text-gray-500">
          {steps.filter(s => s.status === 'completed').length} / {steps.length} 步骤完成
        </div>
      </div>
      
      <div className="space-y-3">
        {steps.map((step, index) => {
          const isActive = index === currentStep;
          const isCompleted = step.status === 'completed';
          const isError = step.status === 'error';
          const isExpanded = expandedSteps.has(index);
          
          return (
            <div 
              key={index} 
              className={`thinking-step border rounded-lg transition-all duration-200 ${
                isActive ? 'border-blue-300 bg-blue-50' : 
                isCompleted ? 'border-green-200 bg-green-50' :
                isError ? 'border-red-200 bg-red-50' :
                'border-gray-200 bg-gray-50'
              } ${mode === 'interactive' ? 'cursor-pointer hover:shadow-sm' : ''}`}
              onClick={() => {
                if (mode === 'interactive') {
                  setSelectedStep(index);
                  onStepClick?.(step, index);
                }
              }}
            >
              <div className="p-3">
                <div className="flex items-start justify-between">
                  <div className="flex items-start space-x-3 flex-1">
                    {/* 状态图标 */}
                    <div className={`w-6 h-6 rounded-full flex items-center justify-center text-sm ${
                      isActive ? 'bg-blue-500 text-white animate-pulse' :
                      isCompleted ? 'bg-green-500 text-white' :
                      isError ? 'bg-red-500 text-white' :
                      'bg-gray-300 text-gray-600'
                    }`}>
                      {isActive ? '⚡' : isCompleted ? '✓' : isError ? '✗' : index + 1}
                    </div>
                    
                    {/* 步骤信息 */}
                    <div className="flex-1">
                      <div className="flex items-center space-x-2">
                        <h4 className="font-medium text-gray-800">{step.title}</h4>
                        {step.confidence !== undefined && (
                          <span className={`text-xs px-2 py-1 rounded-full ${
                            step.confidence > 0.8 ? 'bg-green-100 text-green-800' :
                            step.confidence > 0.6 ? 'bg-yellow-100 text-yellow-800' :
                            'bg-red-100 text-red-800'
                          }`}>
                            {Math.round(step.confidence * 100)}%
                          </span>
                        )}
                      </div>
                      
                      {step.description && (
                        <p className="text-sm text-gray-600 mt-1">{step.description}</p>
                      )}
                      
                      {/* 置信度条 */}
                      {step.confidence !== undefined && renderConfidenceBar(step.confidence)}
                      
                      {/* 处理时间 */}
                      {step.duration && (
                        <div className="text-xs text-gray-500 mt-1">
                          ⏱️ {step.duration}ms
                        </div>
                      )}
                    </div>
                  </div>
                  
                  {/* 展开按钮 */}
                  {showDetails && (step.reasoning || step.alternatives || step.tokenUsage) && (
                    <button
                      onClick={(e) => {
                        e.stopPropagation();
                        toggleStepExpansion(index);
                      }}
                      className="ml-2 p-1 text-gray-400 hover:text-gray-600 transition-colors"
                    >
                      <span className={`transform transition-transform ${
                        isExpanded ? 'rotate-180' : ''
                      }`}>
                        ▼
                      </span>
                    </button>
                  )}
                </div>
                
                {/* 详细信息 */}
                {showDetails && renderThinkingDetails(step, index)}
              </div>
            </div>
          );
        })}
      </div>
      
      {/* 总结统计 */}
      {mode === 'detailed' && (
        <div className="mt-4 pt-4 border-t border-gray-200">
          <div className="grid grid-cols-2 md:grid-cols-4 gap-4 text-center">
            <div className="bg-blue-50 p-3 rounded-lg">
              <div className="text-lg font-semibold text-blue-600">
                {steps.reduce((sum, step) => sum + (step.tokenUsage?.total || 0), 0)}
              </div>
              <div className="text-sm text-blue-500">总Token</div>
            </div>
            <div className="bg-green-50 p-3 rounded-lg">
              <div className="text-lg font-semibold text-green-600">
                {steps.reduce((sum, step) => sum + (step.duration || 0), 0)}ms
              </div>
              <div className="text-sm text-green-500">总耗时</div>
            </div>
            <div className="bg-purple-50 p-3 rounded-lg">
              <div className="text-lg font-semibold text-purple-600">
                {Math.round(steps.reduce((sum, step) => sum + (step.confidence || 0), 0) / steps.length * 100)}%
              </div>
              <div className="text-sm text-purple-500">平均置信度</div>
            </div>
            <div className="bg-orange-50 p-3 rounded-lg">
              <div className="text-lg font-semibold text-orange-600">
                {steps.filter(s => s.status === 'completed').length}
              </div>
              <div className="text-sm text-orange-500">完成步骤</div>
            </div>
          </div>
        </div>
      )}
    </div>
  );
};
```

```typescript
export const UnifiedResultDisplay: React.FC<UnifiedResultDisplayProps> = ({
```
