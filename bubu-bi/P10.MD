export const dataSlice: StateCreator<DataSlice> = (set, get) => ({
  // ... 现有状态
  
  // 计算属性
  get hasData() {
    return get().currentData !== null;
  },
  
  get dataStats() {
    const data = get().currentData;
    if (!data) return null;
    return {
      rowCount: data.rows?.length || 0,
      columnCount: data.columns?.length || 0,
      hasError: !!get().error
    };
  }
});

      const start = performance.now();
      const result = set(...args);
      const end = performance.now();

### 3.2 组件优化 (优先级：🟡 中)

**LazyLoad 组件优化：**
```typescript
// 使用 React 19 的新特性优化
export const LazyLoad = ({ children, ...props }) => {
  const [isVisible, setIsVisible] = useState(false);
  const ref = useRef();
  
  // 使用新的 Intersection Observer 模式
  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setIsVisible(true);
          observer.disconnect();
        }
      },
      { threshold: 0.1 }
    );
    
    if (ref.current) observer.observe(ref.current);
    return () => observer.disconnect();
  }, []);
  
  return (
    <div ref={ref}>
      {isVisible ? children : <Skeleton />}
    </div>
  );
};
```

**PerformanceMonitor 优化：**
```typescript
// 简化性能监控逻辑
export const usePerformanceMonitor = (enabled = false) => {
  const [metrics, setMetrics] = useState(null);
  
  useEffect(() => {
    if (!enabled) return;
    
    const observer = new PerformanceObserver((list) => {
      const entries = list.getEntries();
      setMetrics({
        renderTime: entries[0]?.duration || 0,
        timestamp: Date.now()
      });
    });
    
    observer.observe({ entryTypes: ['measure'] });
    return () => observer.disconnect();
  }, [enabled]);
  
  return metrics;
};
```
